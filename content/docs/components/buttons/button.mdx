## Installation
#### 1. Copy and paste the following code into your project.

<ComponentSource name="button" filePath="@/components/ui/buttons/button.tsx"/>

#### 2. Update *`globals.css`* and add this variables. Adjust the color as you like.
```css
    --primary: 142.1 76.2% 36.3%;
    --primary-foreground: 355.7 100% 97.3%;
    --secondary: 267 100% 60%;
    --secondary-foreground: 277 100% 92%;
    --info: 189 100% 43%;
    --info-foreground: 171 93% 89%;
    --success: 142 71% 45%;
    --success-foreground: 119 88% 91%;
    --warning: 40 100% 50%;
    --warning-foreground: 48 100% 90%;
    --error: 11 100% 59%;
    --error-foreground: 29 100% 92%;
```

#### 3.  Update *`tailwind config`*
```ts
const config: Config = {
  ...
	theme: {
    ...
    	extend: {
        ...
    		colors: {
          ...
    			primary: {
    				DEFAULT: 'hsl(var(--primary) / <alpha-value>)',
    				foreground: 'hsl(var(--primary-foreground) / <alpha-value>)'
    			},
    			secondary: {
    				DEFAULT: 'hsl(var(--secondary) / <alpha-value>)',
    				foreground: 'hsl(var(--secondary-foreground) / <alpha-value>)'
    			},
    			info: {
    				DEFAULT: 'hsl(var(--info) / <alpha-value>)',
    				foreground: 'hsl(var(--info-foreground) / <alpha-value>)'
    			},
    			success: {
    				DEFAULT: 'hsl(var(--success) / <alpha-value>)',
    				foreground: 'hsl(var(--success-foreground) / <alpha-value>)'
    			},
    			warning: {
    				DEFAULT: 'hsl(var(--warning) / <alpha-value>)',
    				foreground: 'hsl(var(--warning-foreground) / <alpha-value>)'
    			},
    			error: {
    				DEFAULT: 'hsl(var(--error) / <alpha-value>)',
    				foreground: 'hsl(var(--error-foreground) / <alpha-value>)'
    			},
    		},
    	}
    },
};

```
